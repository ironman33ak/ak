#include <lpc214x.h>

// Simple delay
void delay(unsigned int d)
{
    unsigned int i, j;
    for(i = 0; i < d; i++)
        for(j = 0; j < 200; j++);
}

int main(void)
{
    unsigned int value;

    // Select P0.25 as DAC output (AOUT)
    PINSEL1 |= (1 << 19);

    while(1)
    {
        // Ramp UP (0 ? 1023)
        for(value = 0; value <= 255; value++)
        {
            DACR = (value << 6);
            delay(1);
        }

        // Sudden drop happens automatically
        // when loop restarts from 0
    }
}
