#include <lpc214x.h>

#define LEDS 0xFF000000

void timer0_delay()
{
    T0TCR = 0x02;

    T0PR  = 14999;

    T0MR0 = 1000;

    T0MCR = 0x03;

    T0TCR = 0x01;

    while(!(T0IR & 0x01));

    T0IR  = 0x01;

    T0TCR = 0x00;
}

int main()
{
    IO1DIR |= LEDS;

    PINSEL2 = 0x00000000;

    while(1)
    {
        IO1SET = LEDS;
        timer0_delay();

        IO1CLR = LEDS;
        timer0_delay();
    }
}
