1. Triangular Wave

#include <lpc214x.h>

void delay(unsigned int t) {
    unsigned int i, j;

    for(i = 0; i < t; i++)
        for(j = 0; j < 3000; j++);
}

int main() {

    PINSEL1 |= (1 << 19);

    while(1) {
        unsigned int i;

        for(i = 0; i < 1024; i++) {
            DACR = (i << 6);
            delay(1);
        }

        for(i = 1023; i > 0; i--) {
            DACR = (i << 6);
            delay(1);
        }
    }
}

3. Sine Wave 
#include <lpc214x.h>

unsigned int sine[32] = {
    512, 612, 707, 793, 866, 924, 963, 991,
    1007, 1007, 991, 963, 924, 866, 793, 707,
    612, 512, 412, 317, 231, 158, 100, 61,
    35, 20, 20, 35, 61, 100, 158, 231
};

void delay(unsigned int t) {
    unsigned int i, j;

    for(i = 0; i < t; i++)
        for(j = 0; j < 1000; j++);
}

int main() {
    unsigned int i;

    PINSEL1 |= (1 << 19);

    while(1) {

        for(i = 0; i < 32; i++) {
            DACR = (sine[i] << 6);
            delay(5);
        }
    }
}
