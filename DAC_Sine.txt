#include <lpc214x.h>

// Approximate sine wave (one cycle, 10-bit values: 0–1023)
unsigned int sine_table[16] = {
    512, 713, 873, 973,
    1023, 973, 873, 713,
    512, 311, 151, 51,
    0,   51, 151, 311
};

// Simple delay
void delay(unsigned int d)
{
    unsigned int i, j;
    for(i = 0; i < d; i++)
        for(j = 0; j < 500; j++);
}

int main(void)
{
    int i;

    // Select P0.25 as DAC output
    PINSEL1 |= (1 << 19);

    while(1)
    {
        for(i = 0; i < 16; i++)
        {
            DACR = (sine_table[i] << 6); // Send value to DAC
            delay(50);
        }
    }
}
