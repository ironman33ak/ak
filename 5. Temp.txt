#include <lpc214x.h>
#include <stdio.h>

#define RS (1 << 11)
#define RW (1 << 20)
#define EN (1 << 10)

#define LEDS 0xFF000000

void delay(unsigned int time) {
    unsigned int i, j;

    for(i = 0; i < time; i++)
        for(j = 0; j < 6000; j++);
}

void lcd_cmd(unsigned char cmd) {
    IO1CLR = 0x00FF0000;
    IO1SET = (cmd << 16);

    IO0CLR = RS | RW;

    IO0SET = EN;
    delay(2);
    IO0CLR = EN;
}

void lcd_data(unsigned char data) {
    IO1CLR = 0x00FF0000;
    IO1SET = (data << 16);

    IO0SET = RS;
    IO0CLR = RW;

    IO0SET = EN;
    delay(2);
    IO0CLR = EN;
}

void lcd_string(char *str) {
    while(*str)
        lcd_data(*str++);
}

void lcd_init() {
    lcd_cmd(0x38);
    lcd_cmd(0x0C);
    lcd_cmd(0x06);
    lcd_cmd(0x01);

    delay(5);
}

void adc_init() {
    PINSEL1 |= (1 << 26);
    PINSEL1 &= ~(1 << 27);

    AD0CR = (1 << 2) | (4 << 8) | (1 << 21);
}

unsigned int adc_read() {
    AD0CR |= (1 << 24);

    while(!(AD0GDR & (1 << 31)));

    return (AD0GDR >> 6) & 0x3FF;
}

int main() {
    unsigned int adc_val, temp;
    char buffer[16];

    IO0DIR |= RS | RW | EN;

    IO1DIR |= 0x00FF0000;

    IO1DIR |= LEDS;

    lcd_init();
    adc_init();

    while(1) {

        adc_val = adc_read();

        temp = (adc_val * 330) / 1023;

        lcd_cmd(0x01);
        lcd_string("Temp:");

        lcd_cmd(0xC0);

        sprintf(buffer, "%d C", temp);

        lcd_string(buffer);

        if(temp > 50) {

            IO1SET = LEDS;
            delay(500);

            IO1CLR = LEDS;
            delay(500);

        } else {

            IO1CLR = LEDS;
        }

        delay(100);
    }
}
