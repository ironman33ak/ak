#include <lpc214x.h>

// Simple delay
void delay(unsigned int d)
{
    unsigned int i, j;
    for(i = 0; i < d; i++)
        for(j = 0; j < 500; j++);
}

int main(void)
{
    unsigned int value;

    // Select P0.25 as DAC output (AOUT)
    PINSEL1 |= (1 << 19);

    while(1)
    {
        // Staircase: increase in steps
        for(value = 0; value <= 1023; value += 128) // step size = 128
        {
            DACR = (value << 6);
            delay(100);   // hold each step
        }

        // Sudden drop back to 0 (new cycle)
        DACR = (0 << 6);
    }
}
