#include <lpc214x.h>

#define RS (1<<11)
#define RW (1<<20)
#define EN (1<<10)

void delay(unsigned int time) {
    unsigned int i, j;

    for(i = 0; i < time; i++)
        for(j = 0; j < 6000; j++);
}

void lcd_cmd(unsigned char cmd) {

    IO1CLR = 0x00FF0000;

    IO1SET = (cmd << 16);

    IO0CLR = RS;
    IO0CLR = RW;

    IO0SET = EN;
    delay(2);

    IO0CLR = EN;
}

void lcd_data(unsigned char data) {

    IO1CLR = 0x00FF0000;

    IO1SET = (data << 16);

    IO0SET = RS;
    IO0CLR = RW;

    IO0SET = EN;
    delay(2);

    IO0CLR = EN;
}

void lcd_init() {

    lcd_cmd(0x38);

    lcd_cmd(0x0C);

    lcd_cmd(0x06);

    lcd_cmd(0x01);

    delay(5);
}

void lcd_string(char *str) {

    while(*str) {
        lcd_data(*str++);
    }
}

int main() {

    IO0DIR |= RS | RW | EN;

    IO1DIR |= 0x00FF0000;

    lcd_init();

    lcd_string("HELLO");

    while(1);
}
