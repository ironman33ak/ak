#include <lpc214x.h>

unsigned char seg_code[10] = {
    0xC0, 0xF9, 0xA4, 0xB0, 0x99,
    0x92, 0x82, 0xF8, 0x80, 0x90
};

void delay() {
    int i, j;
    for(i = 0; i < 1000; i++)
        for(j = 0; j < 1000; j++);
}

int main(void) {
    int i;

    IO1DIR |= 0x00FF0000;

    IO0DIR |= (1 << 10);

    while(1) {
        for(i = 0; i < 10; i++) {

            IO0CLR = (1 << 10);

            IO1CLR = 0x00FF0000;

            IO1SET = (seg_code[i] << 16);

            delay();
        }
    }
}
