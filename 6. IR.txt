#include <lpc214x.h>

#define IR_PIN      (1 << 30)
#define BUZZER_PIN  (1 << 20)

void delay_ms(unsigned int ms)
{
    unsigned int i, j;

    for(i = 0; i < ms; i++)
        for(j = 0; j < 3000; j++);
}

int main(void)
{
    PINSEL0 = 0x00000000;
    PINSEL1 = 0x00000000;

    IO0DIR &= ~IR_PIN;

    IO1DIR |= BUZZER_PIN;

    while(1)
    {
        if(!(IO0PIN & IR_PIN))
        {
            IO1SET = BUZZER_PIN;
            delay_ms(200);

            IO1CLR = BUZZER_PIN;
            delay_ms(200);
        }
        else
        {
            IO1CLR = BUZZER_PIN;
        }
    }
}
